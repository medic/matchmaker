//Adding and removing HTML elements with JS

<div id="div1">
<button onclick="addbutton()">Add a button</button>
</div>

<script>
var i=1;

function addbutton() {
var para=document.createElement("button");
para.setAttribute("id", i);
para.setAttribute("onclick", "removebutton()");

var node=document.createTextNode("Button #" + i);
i++;


para.appendChild(node);

var element=document.getElementById("div1");
element.appendChild(para);
}

function removebutton() {
var child=document.getElementById(event.target.id);
child.parentNode.removeChild(child);
}

</script>
